Functions

Function Syntax


                    function name(parameter1, parameter2, parameter3) {
                        code to be executed
                    }
                

                    let name = function (parameter1, parameter2, parameter3) {
                        code to be executed
                    }

ES2015/ES6


                    let name = (parameter1, parameter2, parameter3) => {
                        code to be executed
                    }
                

                    let name = (parameter1, parameter2) => parameter1 + parameter2;
                

                    let name = parameter1 => parameter1.toString();
                

                    name(new Date());

Complex examples


                    // Returns a function
                    let getRandomGenerator = () => {
                        return () => Math.random();
                    };

                    // Parameter is a function
                    let generate = (generator) => generator();

                    // Call the generate method
                    console.log(generate(getRandomGenerator()));
                    // Call the generate with different generator method
                    console.log(generate(() => Math.ceil(Math.random() * 100)));

Function's this


                    console.log(this === window); // true; Global context

                    function f1() {
                        return this; // window
                    }

                    function f2() {
                        'use strict'; // see strict mode
                        return this; // undefined
                    }

                    f1.call({}, 1, 2); // this is replaced by {}
                    f1.apply({}, [1, 2]); // this is replaced by {}
                    f2.bind({});
                    f2(); // this is replaced by {}

Function's this in objects


                    function Car() {
                        this.f = function() { return this };
                    }

                    let a = new Car();
                    a.f() // this === a

                    function f(x) { return x(); }
                    f(a.f); // this === window / undefined
                    f(a.f.bind(a)); // this === a

ES2015/ES6 - Lexical this


                    this.nums.forEach((v) => {
                        if (v % 5 === 0)
                            this.fives.push(v)
                    })
ES5: 

                    //  variant 1
                    var self = this;
                    this.nums.forEach(function (v) {
                        if (v % 5 === 0)
                            self.fives.push(v);
                    });

                    //  variant 2 (since ECMAScript 5.1 only)
                    this.nums.forEach(function (v) {
                        if (v % 5 === 0)
                            this.fives.push(v);
                    }.bind(this));

ES2015/ES6 - Extended parameter handling


                    function f (x, y = 7, z = 42) { // ES6 default values
                        // ES5: y = y || 7; ...
                        return x + y + z
                    }
                    f(1) === 50
                

                    function f (x, y, ...a) { // ES6 rest parameter
                        // ES5 a = Array.prototype.slice.call(arguments, 2);
                        return (x + y) * a.length
                    }
                    f(1, 2, "hello", true, 7) === 9

Scope and closures

Scope


                    let a = 1; // Global scope: everything registered under
                    // window/script, eg. window.a = 1 (script in case of let)
                    function f1() {
                        let b = 2 + a; // Outer function scope
                        // a is *not* visible here

                        return () => {
                            b += 2;
                            let c = a + b; // Inner function scope
                            return c;
                        };
                        // c is *not* visible here
                    }
                    // b is *not* visible here
                    // c is *not* visible here

Closures

Closures are functions that refer to independent (free) variables (variables that are used locally, but defined in an enclosing scope). In other words, these functions 'remember' the environment in which they were created. 

                    // Global scope: everything registered under window/script,
                    // eg. window.a (script in case of let)
                    let f2 = f1(); // A new closure created!
                    a = f2(); // a and b were modified => window.a = 6, f2.f1.b = 5

                    a = f2(); // a and b were modified => window.a = 13, f2.f1.b = 7

                    let f3 = f1(); // A new closure created! f2.f1 !== f3.f1

Closures


                    let a = 1; // Global scope
                    function f1() {
    let b = 2 + a; // Outer function scope
                        // a is *not* visible here

                        return () => {
        b += 2;
                            let c = a + b; // Inner function scope
                            debugger;
                            return c;
                    
    };
}
                    let f2 = f1();
                    let f3 = f1();

Block-scoped function (ES2015/ES6)


                    {
                        function foo () { return 1 }
                        foo() === 1
                        {
                            function foo () { return 2 }
                            foo() === 2
                        }
                        foo() === 1
                    }

Object.prototype

  • All JavaScript objects inherit the properties and methods from their prototype.
  • Objects created using an object literal, or with new Object(), inherit from a prototype called Object.prototype.
  • The Object.prototype is on the top of the prototype chain.
  • All JavaScript objects (Date, Array, RegExp, Function, ....) inherit from the Object.prototype.

Object.prototype


                    function Person(first, last, age, eye) {
                        ...
                    }

                    Person.prototype.nationality = 'English';

                    Person.prototype.name = function() {
                        return this.firstName + ' ' + this.lastName;
                    };

                    let myFather = new Person('John', 'Doe', 50, 'blue');
                    console.log(myFather.name());

Prototype inheritance


                    function Teacher(first, last, age, gender, interests, subject) {
                        Person.call(this, first, last, age, gender, interests);
                        this.subject = subject;

                    Teacher.prototype = Object.create(Person.prototype);
                    Teacher.prototype.constructor = Teacher;

                    Teacher.prototype.greeting = function() { ... }

Getters and setters

                        constructor (width, height) {
                            this._width  = width
                            this._height = height

                        set width  (width)  { this._width = width               }

                        get area   ()       { return this._width * this._height }

  • Can have also non-numeric keys → not part of the array

instanceof


                    function car () {}
                    let a = new car
                    a instanceof car                            // true
                    a instanceof car.prototype.constructor      // true
                    a.constructor === car.prototype.constructor // true

                    car.prototype.constructor = function () {}
                    a instanceof car.prototype.constructor      // false
                    a instanceof car                            // true

                    a.constructor = function() {}
                    a instanceof car                            // true

Expressions

  • JavaScript evaluates expressions from left to right. Different sequences can produce different results.
  • Values are automatically converted if needed (for all operators except === and !==)
  • Avoid type conversions as much as possible → slow and unpredictable

Expressions


                    let x = 16 + 4 + 'Volvo';
                    // x → 20Volvo
                    x = 'Volvo' + 16 + 4;
                    // x → Volvo164
                    x = 'Volvo';
                    let y = 8 - x;
                    // y → NaN
                    x = 4;
                    y = 8 - x;
                    // y → 4
                    x = '5' + 4 // → 54
                    x = '5' - 4 // → 1

Booleans and conversion

  • Everything With a "Real" Value is True

                    if('a')         // → true
                    if(1)           // → true
                    if(-1)          // → true
                    if(Infinity)    // → true
                    if({})          // → true
                    if([])          // → true

                    if('')          // → false
                    if(0)           // → false
                    if(NaN)         // → false
                    if(null)        // → false
                    if(undefined)   // → false

Booleans and conversion


                    let x = ...;

                    if(!x) { // Evaluates to true if x is convertible to false
                        x = ...;
                    }

                    // Logical "or" evaluates to the first operand that is convertible
                    // to true
                    x = x || {}; // Evaluates to {} if x is convertible to false

                    // To check for null/undefined only
                    // Only undefined can be converted to null
                    if(x != null) {  // x == null
                        ...
                    }

                    let str = 'foo'
